Instacart Analysis

library(tidyverse)
library(viridis)
library(p8105.datasets)
library(plotly)
set.seed(1)

data("instacart")
nyc_instacart = 
  instacart %>% 
  select(
    user_id, reordered, product_name, order_hour_of_day, aisle) %>% 
  sample_n(5000)
nyc_instacart %>% 
  group_by (product_name, reordered, aisle) %>%
  summarise(n = n()) %>%
  filter(n>5) %>% 
  plot_ly(x = ~ n, y = ~ product_name, type = "scatter", mode = "markers", color= ~aisle, alpha = 0.5)
